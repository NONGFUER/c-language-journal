ADT List {
    数据对象：D = {ai | ai∈ElemSet, i=1,2,...,n, n≥0}
    数据关系：R1 = {<ai-1, ai> | ai-1, ai∈D, i=2,3,...,n}
    基本操作：
        InitList(&L) //线性表的初始化
        DestroyList(&L)
        ClearList(&L)
        ListEmpty(L)
        ListLength(L)
        GetElem(L, i, &e)
        LocateElem(L, e,compare())
        PriorElem(L, cur_e, &pre_e)
        NextElem(L, cur_e, &next_e)
        ListInsert(&L, i, e)
        ListDelete(&L, i, &e)
        TraverseList(L, visit())
} ADT List


InitList(&L) :构造一个空的线性表L

DestroyList(&L) :销毁线性表L

ClearList(&L) :清空线性表L

ListEmpty(L) :判断线性表L是否为空

ListLength(L) :返回线性表L的元素个数

GetElem(L, i, &e) :返回线性表L中第i个位置的元素e

LocateElem(L, e,compare()) :返回线性表L中第一个值为e的元素的位置

priorElem(L, cur_e, &pre_e) :返回线性表L中元素cur_e的直接前驱

NextElem(L, cur_e, &next_e) :返回线性表L中元素cur_e的直接后继

ListInsert(&L, i, e) :在线性表L中第i个位置插入元素e

ListDelete(&L, i, &e) :在线性表L中删除第i个位置的元素，并返回该元素

ListTraverse(L, visit()) :依次对线性表L中的每个元素调用visit()函数
